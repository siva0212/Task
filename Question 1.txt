// To check the Phrase/String is palindrome or not
bool bFnPalindrome(string str)
{
    int left = 0, right =0;

    // convert to lower case string
    for (int i = 0; i < str.length(); i++)
        str[i] = tolower(str[i]);


    right = str.length() - 1;


    // Compares character by character
    while (left <= right) {

        // Check if there is symbol in left side
        if (!(str[left] >= 'a' && str[left] <= 'z'))
            left++;

        // Check if there is symbol in right side

        else if (!(str[right] >= 'a' && str[right] <= 'z'))
            right--;

        // Check characters are equal
        else if (str[left] == str[right])
        {
            left++;
            right--;

        }


        else
           // if not equal means the phrase is not palindrome
            return false;
    }

    // Returns true if phrase is palindrome
    return true;
}
